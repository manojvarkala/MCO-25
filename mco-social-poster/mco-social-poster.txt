if (!defined('ABSPATH')) exit;

define('MCO_SOCIAL_PLUGIN_DIR', plugin_dir_path(__FILE__));

require_once MCO_SOCIAL_PLUGIN_DIR . 'includes/admin-page.php';
require_once MCO_SOCIAL_PLUGIN_DIR . 'includes/post-handler.php';

// Add action links
function mco_social_add_action_links($links) {
    $settings_link = '<a href="' . admin_url('admin.php?page=mco-social-poster') . '">Settings</a>';
    array_unshift($links, $settings_link);
    return $links;
}
add_filter('plugin_action_links_' . plugin_basename(__FILE__), 'mco_social_add_action_links');
