// === CENTRAL FORM SUBMISSION HANDLER (HOOKED TO admin_init) ===
if (!function_exists('mco_handle_form_submissions')) {
    function mco_handle_form_submissions() {
        // Handle CSV Uploads
        if (isset($_POST['mco_exam_csv_upload_nonce']) && check_admin_referer('mco_exam_csv_upload_action', 'mco_exam_csv_upload_nonce')) {
            mco_handle_exam_csv_upload();
        }
        if (isset($_POST['mco_book_csv_upload_nonce']) && check_admin_referer('mco_book_csv_upload_action', 'mco_book_csv_upload_nonce')) {
            mco_handle_book_csv_upload();
        }

        // Handle Tools Tab Actions
        if (isset($_POST['mco_flush_rewrites']) && check_admin_referer('mco_flush_rewrites_nonce')) {
            flush_rewrite_rules(true);
            add_action('admin_notices', function() {
                echo '<div class="notice notice-success is-dismissible"><p>API routes have been successfully refreshed.</p></div>';
            });
        }
    }
}

if (!function_exists('mco_auto_update_config_version_on_save')) {
    function mco_auto_update_config_version_on_save($post_id) {
        if (wp_is_post_revision($post_id)) return;
        $post_type = get_post_type($post_id);
        if (in_array($post_type, ['mco_exam_program', 'mco_recommended_book', 'product'])) {
            mco_update_config_version();
        }
    }
}

if (!function_exists('mco_handle_exam_csv_upload')) {
    function mco_handle_exam_csv_upload() {
        if (!current_user_can('manage_options') || empty($_FILES['exam_program_csv']['tmp_name'])) return;

        $file = $_FILES['exam_program_csv']['tmp_name'];
        $handle = fopen($file, "r");
        $header = fgetcsv($handle);
        if ($header === false) { fclose($handle); return; }
        $header = array_map('trim', $header);

        $imported = 0; $updated = 0; $errors = []; $row_num = 1;

        while (($row = fgetcsv($handle)) !== FALSE) {
            $row_num++;
            if (count($header) !== count($row)) { $errors[] = "Row {$row_num}: Mismatched column count."; continue; }
            $data = array_combine($header, $row);
            if ($data === false) { $errors[] = "Row {$row_num}: Failed to process row."; continue; }

            $title = sanitize_text_field($data['program_title']);
            if (empty($title)) { $errors[] = "Row {$row_num}: Empty title."; continue; }
            
            $existing_post_query = new WP_Query(['post_type' => 'mco_exam_program', 'title' => $title, 'post_status' => 'any', 'posts_per_page' => 1]);
            
            $post_id = 0;
            if ($existing_post_query->have_posts()) {
                $post_id = $existing_post_query->posts[0]->ID;
                wp_update_post(['ID' => $post_id, 'post_content' => wp_kses_post($data['program_description'])]);
                $updated++;
            } else {
                $post_status = sanitize_text_field($data['status']);
                if (!in_array($post_status, ['publish', 'draft', 'pending', 'private'])) $post_status = 'publish';
                $post_id = wp_insert_post(['post_title' => $title, 'post_content' => wp_kses_post($data['program_description']), 'post_type' => 'mco_exam_program', 'post_status' => $post_status], true);
                if (is_wp_error($post_id)) { $errors[] = "Row {$row_num}: Failed to create '{$title}'. Error: " . $post_id->get_error_message(); continue; }
                $imported++;
            }
            wp_reset_postdata();

            if ($post_id > 0) {
                update_post_meta($post_id, '_mco_question_source_url', esc_url_raw($data['question_source_url']));
                update_post_meta($post_id, '_mco_certification_exam_sku', sanitize_text_field($data['certification_exam_sku']));
                update_post_meta($post_id, '_mco_is_proctored', absint($data['is_proctored']));
                if (!empty($data['recommended_book_id'])) update_post_meta($post_id, '_mco_recommended_book_id', absint($data['recommended_book_id']));
                
                wp_set_post_terms($post_id, strval(absint($data['practice_questions'])), 'exam_practice_questions');
                wp_set_post_terms($post_id, strval(absint($data['practice_duration'])), 'exam_practice_duration');
                wp_set_post_terms($post_id, strval(absint($data['cert_questions'])), 'exam_cert_questions');
                wp_set_post_terms($post_id, strval(absint($data['cert_duration'])), 'exam_cert_duration');
                wp_set_post_terms($post_id, strval(absint($data['pass_score'])), 'exam_pass_score');
            }
        }
        fclose($handle);
        mco_update_config_version();

        add_action('admin_notices', function() use ($imported, $updated, $errors) {
            echo '<div class="notice notice-success is-dismissible"><p><strong>Exam Import Complete!</strong> ';
            echo "Created {$imported} new programs. Updated {$updated} existing programs.</p></div>";
            if (!empty($errors)) {
                echo '<div class="notice notice-error"><p><strong>Import Errors:</strong></p><ul style="list-style-type:disc; padding-left:20px;">';
                foreach ($errors as $error) echo '<li>' . esc_html($error) . '</li>';
                echo '</ul></div>';
            }
        });
    }
}

    function mco_sanitize_certificate_templates($input) {
        $output = [];
        if (is_array($input)) {
            foreach ($input as $id => $template) {
                if (is_array($template)) {
                    $sanitized_id = sanitize_key($id);
                    $output[$sanitized_id] = [
                        'id' => $sanitized_id,
                        'title' => sanitize_text_field($template['title']),
                        'body' => wp_kses_post($template['body']),
                        'signature1Name' => sanitize_text_field($template['signature1Name']),
                        'signature1Title' => sanitize_text_field($template['signature1Title']),
                        'signature1ImageUrl' => mco_sanitize_signature_field($template['signature1ImageUrl']),
                        'signature2Name' => sanitize_text_field($template['signature2Name']),
                        'signature2Title' => sanitize_text_field($template['signature2Title']),
                        'signature2ImageUrl' => mco_sanitize_signature_field($template['signature2ImageUrl']),
                    ];
                }
            }
        }
        mco_update_config_version();
        return $output;
    }
