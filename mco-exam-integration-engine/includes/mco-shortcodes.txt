if (!function_exists('mco_generate_exam_jwt')) {
    // This function is usually provided by mco-api.php.
    // This fallback would only be used if mco-api.php wasn't loaded or didn't define it.
    // It's a minimal stub, not a full JWT implementation for safety.
    function mco_generate_exam_jwt($user_id) {
        // Fallback for when mco-api.php's full JWT generation isn't available
        error_log('MCO SHORTCODE: Using JWT fallback, full API not loaded.');
        if (!defined('MCO_JWT_SECRET')) return null;

        $user = get_user_by('ID', $user_id);
        if (!$user) return null;

        $header = base64_encode(json_encode(['typ' => 'JWT', 'alg' => 'HS256']));
        $payload = base64_encode(json_encode([
            'iss' => get_bloginfo('url'),
            'iat' => time(),
            'exp' => time() + (14 * DAY_IN_SECONDS), // Short expiry for fallback
            'user' => [
                'id' => (string)$user->ID,
                'name' => $user->display_name,
                'email' => $user->user_email,
                'isAdmin' => user_can($user->ID, 'manage_options')
            ],
            'paidExamIds' => [], // Empty for fallback
            'isSubscribed' => false,
            'subscriptionInfo' => null,
            'isBetaTester' => false
        ]));
        
        $signature = hash_hmac('sha256', "{$header}.{$payload}", MCO_JWT_SECRET, true);
        return rtrim(strtr($header, '+/', '-_'), '=') . "." . rtrim(strtr($payload, '+/', '-_'), '=') . "." . rtrim(strtr(base64_encode($signature), '+/', '-_'), '=');
    }
}

if (!function_exists('mco_load_shortcode_css')) {
    function mco_load_shortcode_css() {
        $css_path = MCO_PLUGIN_PATH . 'assets/mco-styles.css';
        if (file_exists($css_path)) {
            $css = file_get_contents($css_path);
            if ($css) {
                // Minify CSS by removing comments and extra whitespace
                $css = preg_replace('!/\*[^*]*\*+([^/][^*]*\*+)*/!', '', $css);
                $css = str_replace(array("\r\n", "\r", "\n", "\t", '  ', '    ', '    '), '', $css);
                echo '<style>' . $css . '</style>';
                return;
            }
        }
        echo mco_get_scoped_css();
    }
}

if (!function_exists('mco_get_hash_of_string_php')) {
    function mco_get_hash_of_string_php($str) {
        $hash = 0;
        if (empty($str)) return $hash;
        $len = strlen($str);
        // Using a large prime for modulo to prevent overflow before final result.
        // This effectively keeps the hash within a manageable integer range,
        // preventing it from turning into a float.
        $prime_mod = 2147483647; // A large prime, which fits within a 32-bit signed integer.
        
        for ($i = 0; $i < $len; $i++) {
            $char_code = ord($str[$i]);
            // Perform arithmetic, then modulo to prevent overflow.
            // Ensure the result remains positive after modulo if intermediate result was negative.
            $hash = ($hash * 31 + $char_code) % $prime_mod;
            if ($hash < 0) { // Ensure hash is always positive
                $hash += $prime_mod;
            }
        }
        return $hash; // This will always be a positive integer within $prime_mod
    }
}

// --- GEO-CODING HELPER (PHP Native) ---
if (!function_exists('mco_get_geo_primary_link_info')) {
    function mco_get_geo_primary_link_info($links, $override_geo_key = null) {
        if (empty($links) || !is_array($links)) return null;
        
        // Filter out empty or whitespace-only links
        $links = array_filter(array_map('trim', $links)); 
        if (empty($links)) return null;

        $domainMap = [
            'com' => 'Amazon.com', 
            'in' => 'Amazon.in', 
            'ae' => 'Amazon.ae'
        ];
        
        $chosen_key = null;

        // 1. Prioritize override geo key from URL parameter (React app navigation)
        if (!empty($override_geo_key) && isset($links[$override_geo_key]) && !empty($links[$override_geo_key])) {
            $chosen_key = $override_geo_key;
        } 
        
        // 2. If no override from URL, check for client-side preference via cookie
        if (empty($chosen_key) && isset($_COOKIE['mco_wp_geo_pref'])) {
            $cookie_geo_key = sanitize_text_field($_COOKIE['mco_wp_geo_pref']);
            if (isset($links[$cookie_geo_key]) && !empty($links[$cookie_geo_key])) {
                $chosen_key = $cookie_geo_key;
            }
        }

        // 3. Fallback to React app's default priority if no explicit key is found
        if (empty($chosen_key)) {
            $fallback_priority = ['com', 'in', 'ae']; // Matches React app's fallback order
            foreach ($fallback_priority as $key) {
                if (isset($links[$key]) && !empty($links[$key])) {
                    $chosen_key = $key;
                    break;
                }
            }
        }
        
        if (!empty($chosen_key)) {
            return [ 
                'key' => $chosen_key, 
                'url' => $links[$chosen_key], 
                'domainName' => $domainMap[$chosen_key] ?? 'Amazon.com' 
            ];
        }
        
        return null; // No valid links found
    }
}

// --- LOGIN SHORTCODE ---
if (!function_exists('mco_exam_login_shortcode')) {
    function mco_exam_login_shortcode() {
        if (is_user_logged_in() && isset($_POST['mco_update_name_nonce']) && wp_verify_nonce($_POST['mco_update_name_nonce'], 'mco_update_name_action')) {
            $full_name = sanitize_text_field($_POST['full_name']);
            if (!empty($full_name) && strpos(trim($full_name), ' ') === false) { // FIX: Only redirect if full name is valid
                return mco_render_name_update_form('Please enter your full first and last last name.');
            }
            if (!empty($full_name)) { // Proceed only if full_name is provided
                $user_id = get_current_user_id();
                wp_update_user(['ID' => $user_id, 'display_name' => $full_name]);
                $name_parts = explode(' ', $full_name, 2);
                update_user_meta($user_id, 'first_name', $name_parts[0]);
                if (isset($name_parts[1])) update_user_meta($user_id, 'last_name', $name_parts[1]);
                return mco_redirect_to_app_with_token();
            } else {
                return mco_render_name_update_form('Please enter your full first and last name.');
            }
        }

        if (is_user_logged_in()) {
            $user = wp_get_current_user();
            if (strpos(trim($user->display_name), ' ') === false) { // No space means it's likely not a full name
                return mco_render_name_update_form();
            }
            return mco_redirect_to_app_with_token();
        }

        $login_url = wp_login_url(get_permalink());
        if (!headers_sent()) { wp_redirect($login_url); exit; }
        mco_load_shortcode_css();
        return '<script>window.location.href="' . esc_url($login_url) . '";</script><div id="mco-app-root"><div class="mco-form-container" style="text-align:center;"><p>Redirecting to login...</p><a href="' . esc_url($login_url) . '" class="mco-button">Click here if not redirected</a></div></div>';
    }
}

        // Ensure mco_render_procedural_cover and mco_get_svg_icon are defined
        if (!function_exists('mco_render_procedural_cover_full') || !function_exists('mco_get_svg_icon')) {
            return '<p>Application is not configured correctly. Missing core functions.</p>';
        }

        $app_program_id = $program_data['id'] ?? ('prod-' . ($program_data['ID'] ?? ''));
        $program_url = rtrim($app_url, '/') . '/program/' . $app_program_id;
        
        $thumbnail_url = $program_data['thumbnailUrl'] ?? '';
        
        $coverHtml = !empty($thumbnail_url) 
            ? '<img src="' . esc_url($thumbnail_url) . '" style="width:100%;height:100%;object-fit:cover;" crossorigin="' . (str_starts_with($thumbnail_url, 'data:image') ? '' : 'anonymous') . '">'
            : mco_render_procedural_cover_full(['title' => $program_data['title']], 'program'); // Simplified to let React handle procedural covers
        
        $desc = wp_trim_words(html_entity_decode($program_data['description'] ?? '', ENT_QUOTES, 'UTF-8'), 20, '...');
        
        return '
        <div class="mco-book-card">
            <div class="mco-book-cover">' . $coverHtml . '</div>
            <div class="mco-book-card__body">
                <a href="' . esc_url($program_url) . '" style="text-decoration:none;"><h4 class="mco-book-card__title">' . esc_html(html_entity_decode($program_data['title'] ?? '', ENT_QUOTES, 'UTF-8')) . '</h4></a>
                <p class="mco-book-card__desc">' . esc_html($desc) . '</p>
            </div>
            <div class="mco-book-card__footer">
                <div class="mco-store-buttons">
                    <a href="' . esc_url($program_url) . '" class="mco-book-btn mco-book-btn--primary">
                        View Program ' . mco_get_svg_icon('external-link') . '
                    </a>
                </div>
            </div>
        </div>';
    }
}

if (!function_exists('mco_single_book_shortcode')) {
    function mco_single_book_shortcode() {
        global $post;
        // Ensure mco_get_app_config_data is defined
        if (!function_exists('mco_get_app_config_data')) {
            return '<p>Application is not configured correctly. Missing core functions.</p>';
        }

        $config_data = mco_get_app_config_data();
        $book_id_meta = get_post_meta($post->ID, '_mco_book_id', true);
        $book = current(array_filter($config_data['suggestedBooks'] ?? [], function($b) use ($book_id_meta) { return ($b['id'] ?? '') === $book_id_meta; }));

        if (!$book) {
            // If book not found in config_data, create a fallback structure from post data
            $book = [
                'id' => $book_id_meta ?: 'book-'.$post->ID, // Ensure an ID
                'title' => get_the_title($post),
                'description' => $post->post_content,
                'thumbnailUrl' => get_post_meta($post->ID, '_mco_thumbnail_url', true) ?: get_the_post_thumbnail_url($post->ID, 'large'),
                'permalink' => get_permalink($post->ID),
                'affiliateLinks' => [
                    'com' => get_post_meta($post->ID, '_mco_link_com', true),
                    'in' => get_post_meta($post->ID, '_mco_link_in', true),
                    'ae' => get_post_meta($post->ID, '_mco_link_ae', true),
                ]
            ];
        }
        return mco_render_single_book_content($book, $post->post_content);
    }
}

if (!function_exists('mco_render_single_program_content')) {
    function mco_render_single_program_content($program_data_arr) {
        if (!$program_data_arr) return '';
        ob_start();
        mco_load_shortcode_css();
        
        $program_data_arr['title'] = html_entity_decode($program_data_arr['title'] ?? '', ENT_QUOTES, 'UTF-8');
        $program_data_arr['description'] = html_entity_decode($program_data_arr['description'] ?? '', ENT_QUOTES, 'UTF-8');
        $program_data_arr['thumbnailUrl'] = $program_data_arr['thumbnailUrl'] ?? '';
        $program_data_arr['id'] = $program_data_arr['id'] ?? ('prod-' . ($program_data_arr['ID'] ?? ''));

        $program_json = json_encode($program_data_arr);
        $unique_id = 'mco-single-program-root-' . uniqid();

        ?>
        <div id="mco-app-root">
            <div class="mco-wrapper">
                <div id="<?php echo $unique_id; ?>">
                    <p class="mco-loading-text">Loading program details...</p>
                </div>
            </div>
        </div>

        <script type="module">
            (async () => {
                // Ensure React and ReactDOM are available globally
                if (typeof window.React === 'undefined' || typeof window.ReactDOM === 'undefined') {
                    const loadScript = (src) => new Promise((resolve, reject) => {
                        const script = document.createElement('script');
                        script.src = src;
                        script.type = 'module';
                        script.onload = resolve;
                        script.onerror = reject;
                        document.head.appendChild(script);
                    });

                    try {
                        await Promise.all([
                            loadScript('https://esm.sh/react@18.2.0?bundle'),
                            loadScript('https://esm.sh/react-dom@18.2.0/client?bundle')
                        ]);
                        console.log("React and ReactDOM loaded dynamically for single program.");
                    } catch (error) {
                        console.error("Failed to load React/ReactDOM dynamically for single program:", error);
                        const container = document.getElementById('<?php echo $unique_id; ?>');
                        if (container) {
                            container.innerHTML = '<p style="color:red;">Error: React rendering components could not be loaded. Check console for details.</p>';
                        }
                        return;
                    }
                }

                const appBaseUrl = window.mcoReactAppBaseUrl || '<?php echo esc_url(mco_get_exam_app_url()); ?>';
                const componentPath = `${appBaseUrl.replace(/\/$/, '')}/dist/assets/ProgramRenderer.js`; // FIX: Changed path to /dist/assets/

                try {
                    const module = await import(componentPath);
                    const ProgramRenderer = module.default;
                    const container = document.getElementById('<?php echo $unique_id; ?>');
                    if (container) {
                        const props = JSON.parse(container.dataset.props || '{}');
                        const root = window.ReactDOM.createRoot(container);
                        root.render(React.createElement(ProgramRenderer, { program: props }));
                    }
                } catch (error) {
                    console.error("Failed to load or render ProgramRenderer (single program):", error);
                    const container = document.getElementById('<?php echo $unique_id; ?>');
                    if (container) {
                        container.innerHTML = `<p style="color:red;">Error loading single program: ${error.message}. Please check console for details.</p>`;
                    }
                }
            })();
        </script>
        <div data-props='<?php echo esc_attr($program_json); ?>' id="<?php echo $unique_id; ?>"></div>
        <?php
        return ob_get_clean();
    }
}

// --- AUTO INJECT TEMPLATE ---
if (!function_exists('mco_auto_inject_template')) {
    add_filter('the_content', 'mco_auto_inject_template');

    function mco_auto_inject_template($content) {
        static $css_loaded = false;

        if (is_singular('mco_recommended_book') && in_the_loop() && is_main_query()) {
            return mco_single_book_shortcode();
        }
        if (is_singular('mco_exam_program') && in_the_loop() && is_main_query()) {
            global $post;
            // Ensure mco_get_app_config_data is defined
            if (!function_exists('mco_get_app_config_data')) {
                return '<p>Application is not configured correctly. Missing core functions.</p>';
            }

            $config_data = mco_get_app_config_data();
            $app_url = mco_get_exam_app_url();

            $app_program_id = 'prod-' . $post->ID;
            $program_category = current(array_filter($config_data['examProductCategories'], function($c) use ($app_program_id) { return $c['id'] === $app_program_id; }));
            
            // If category is not found, skip rendering to prevent errors
            if (!$program_category) {
                error_log("MCO SHORTCODE: Program category not found for post ID: {$post->ID}. Skipping rendering.");
                return ''; 
            }

            $item_data = [
                'id' => $app_program_id, // Pass WP Post ID for thumbnail fallback
                'title' => $post->post_title,
                'description' => $post->post_content,
                'thumbnailUrl' => $program_category['thumbnailUrl'] ?? '', // Use pre-fetched thumbnail from app_config_data
                'permalink' => rtrim($app_url, '/') . '/program/' . $app_program_id,
            ];

            $card_html = mco_render_program_card($item_data, $app_url);
            
            // Ensure CSS is loaded, but only once per page load to avoid bloat
            // We capture it into a variable to prepend to the first card
            $styles = '';
            if (!$css_loaded) {
                ob_start();
                mco_load_shortcode_css();
                $styles = ob_get_clean();
                $css_loaded = true;
            }
            
            return $styles . $card_html;
        }
        return $content;
    }
}

if (!function_exists('mco_redirect_to_app_with_token')) {
    function mco_redirect_to_app_with_token() {
        // Ensure mco_generate_exam_jwt and mco_get_exam_app_url are defined (from mco-api.php)
        if (!function_exists('mco_generate_exam_jwt') || !function_exists('mco_get_exam_app_url')) {
            error_log('MCO SHORTCODE REDIRECT ERR: Critical API functions not defined. Check mco-api.php.');
            return '<p>Error: Application is not configured correctly. Missing core functions.</p>';
        }

        $token = mco_generate_exam_jwt(get_current_user_id());
        if (!$token) {
            return '<p>Error: Could not generate authentication token.</p>';
        }
        $app_url = mco_get_exam_app_url();
        if (!$app_url) return '<p>Error: App URL not configured.</p>';
        
        // FIX: Include the dynamic API URL in the redirect so the React app knows where to connect back to.
        $site_url = get_site_url();
        $redirect_url = add_query_arg(['token' => $token, 'api_url' => $site_url], rtrim($app_url, '/') . '/auth');
        
        nocache_headers();
        wp_redirect($redirect_url);
        exit;
    }
}

/**
 * --- GLOBAL JS VARIABLE INJECTION ---
 * Injects the React App's base URL into the global JavaScript scope.
 * This is crucial for dynamically loading React components within shortcodes,
 * allowing them to locate their built assets relative to the app's root.
 */
if (!function_exists('mco_set_react_app_base_url_for_js')) {
    function mco_set_react_app_base_url_for_js() {
        $app_url = mco_get_exam_app_url();
        if (!empty($app_url)) {
            // Ensure the URL does not end with a slash for consistent path construction
            echo '<script>window.mcoReactAppBaseUrl = "' . rtrim(esc_url($app_url), '/') . '";</script>' . "\n";
        }
    }
    // Use wp_footer to ensure mco_get_exam_app_url() has been evaluated in the request.
    add_action('wp_footer', 'mco_set_react_app_base_url_for_js');
}

/**
 * --- INJECT CLIENT-SIDE GEO PREFERENCE SCRIPT ---
 * Reads localStorage.mco_preferred_geo_key (set by React app) and saves it to a session cookie.
 * This allows PHP to retrieve the client's geo preference on direct WordPress page loads.
 */
if (!function_exists('mco_inject_geo_preference_script')) {
    function mco_inject_geo_preference_script() {
        ?>
        <script>
            (function() {
                const preferredGeoKey = localStorage.getItem('mco_preferred_geo_key');
                if (preferredGeoKey) {
                    // Set a session cookie (expires when browser closes)
                    document.cookie = `mco_wp_geo_pref=${preferredGeoKey}; path=/; SameSite=Lax`;
                } else {
                    // Clear the cookie if no preference is stored in localStorage
                    document.cookie = `mco_wp_geo_pref=; Max-Age=0; path=/; SameSite=Lax`;
                }
            })();
        </script>
        <?php
    }
    add_action('wp_footer', 'mco_inject_geo_preference_script');
}
